\begin{tikzpicture}
    \pgfplotsset{
        compat=1.10,
        width=\getDarcImageFactor * 16cm,
        standard/.style={
                axis x line=middle,
                axis y line=left,
                ticklabel style={fill=white},
                set layers=tick labels on top% use layers and choose the new layer set
            },
        layers/tick labels on top/.define layer set=% define the new layer set based on the standard one
            {axis background,axis grid,axis ticks,axis lines,main,%
                axis tick labels,% <- tick labels before main
                axis descriptions,axis foreground}
            {/pgfplots/layers/standard},
        clip=false
    }

    \begin{polaraxis}[
        /pgf/number format/.cd,
        use comma,
        domain=-180:180,
        ymin=-0.2,
        ymax=1.05,
        xmin=-45,
        xmax=45,
        grid=none,
        axis line style={draw=none},
        ticks=none,
        ytick={0.3,0.4,0.5,0.6,0.7,0.8,0.9,1},
        yticklabels={,,},
        xtick={-40,-30,-20,-10,0,10,20,30,40},
        xticklabels={}
    ]

        \draw[very thick, DARCblue] plot[
                variable=\t,
                domain=0:360,
                smooth,samples=101
            ]
        ({40*sin(\t)}:{129*pow(0.95*sin(\t/2),5)});

        \draw[very thick, DARCblue] plot[
                variable=\t,
                domain=0:360,
                smooth,samples=101
            ]
        ({180+40*sin(\t)}:{12*pow(sin(\t/2),9)});

        \draw[anchor=base] (0,0) node[](origin){};

        \path(0:0) -- ++(-40:100) coordinate (E1);
        \path(0:0) -- ++(0:110) coordinate (F1);
        \path(0:0) -- ++(180:30) coordinate (G1);
    \end{polaraxis}

    \draw[very thick]
    (origin) ++(0,0) -- +(1.5,0)
    (origin) ++(0,-0.8)   -- +(0, 1.6)
    (origin) ++(0.3,-0.7) -- +(0.0, 1.4)

    (origin) ++(1.2,-.4)  -- +(0.0, 0.8)
    (origin) ++(1.5,-.3)  -- +(0.0, 0.6);
    \draw[very thick] (origin) ++(0.15,0) coordinate(E2) -- (E2|-E1) coordinate (E3);
    \draw[very thick] (E3) -- (E3-|F1) coordinate (F2);
    \draw[very thick] (E3) -- (E3-|G1) coordinate (G2);
